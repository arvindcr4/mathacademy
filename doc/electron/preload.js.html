Notification API

 
  notifications: {
    isSupported: () => Notification.isSupported(),
    show: (title, options = {}) => {
      if (!Notification.isSupported()) {
        return false;
      }

      const notification = new Notification({
        title,
        body: options.body || "",
        icon: options.icon,
        silent: options.silent || false,
      });

      notification.on("click", () => {
        notification.close();
        if (options.onClick) {
          options.onClick();
        }
      });

      notification.show();
      return true;
    },
  },
